    """
    Implements the merge sort algorithm.

    Args:
        arr: The list to be sorted.

    Returns:
        The sorted list.
    """
    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2
    left = arr[:mid]
    right = arr[mid:]

    left_merge(left, right, [])
    right_merge(right, [])

    return merge(left, right, [])

    while i < len(left):
